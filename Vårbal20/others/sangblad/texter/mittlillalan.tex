\begin{song}{Mitt lilla lån}{mittlillalan}

\repopen Mitt lilla lån det räcker inte\\

Och till en kursbok emellanåt\\
Mitt lilla lån det räcker inte\\
Det går till öl och till brännvin\\
\end{vers}
\end{song}
